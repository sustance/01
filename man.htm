Usage: omxplayer [OPTIONS] [FILE]
-h --help             print this help
-n --aidx  index      audio stream index:eg 1
-o --adev  device     audio out dev hdmi/local
-i --info             dump stream format, exit
-s --stats            pts and buffer stats
-p --passthrough      audio passthrough
-d --deinterlace      deinterlacing
-w --hw               hw audio decoding
-3 --3d mode          switch tv  eg SBS/TB
-y --hdmiclocksync    refresh rate match video
-z --nohdmiclocksync  not refresh rate match 
-t --sid index        show subtitle with index
-r --refresh          adjust rate/res to video
-l --pos              start position (in secs)
   --boost-on-downmix boost volume w' downmix
   --subtitles path   external subs srt fmt
   --font path        subtitle font
   --font-size size   000's screen height *55
   --align left/center subtitle alignmen *left
   --lines n          in the subtitle buffer*3

z           Show Info
1           Increase Speed
2           Decrease Speed
j           Previous Audio stream
k           Next Audio stream
i           Previous Chapter
o           Next Chapter
n           Previous Subtitle stream
m           Next Subtitle stream
s           Toggle subtitles
d           Subtitle delay -250 ms
f           Subtitle delay +250 ms
q           Exit OMXPlayer
Space or p  Pause/Resume
-           Decrease Volume
+           Increase Volume
Left Arrow  Seek -30
Right Arrow Seek +30
Down Arrow  Seek -600
Up Arrow    Seek +600
